<script>
const totalPages=5;let currentPage=1,isAnimating=false,starsState=JSON.parse(localStorage.getItem('sm_stars')||'[]');
document.addEventListener('DOMContentLoaded',()=>{buildNavDots();buildStars();updateStarsUI()});
function buildNavDots(){const c=document.getElementById('navDots');for(let i=1;i<=totalPages;i++){const d=document.createElement('div');d.className=`nav-dot ${i===1?'active':''}`;d.onclick=()=>goToPage(i);c.appendChild(d)}}
function buildStars(){const g=document.getElementById('starsGrid');for(let i=1;i<=30;i++){const c=document.createElement('div'),col=starsState.includes(i);c.className=`star-cell flex flex-col items-center gap-1 p-2 sm:p-3 rounded-xl border border-white/[0.04] bg-white/[0.01] ${col?'colored bg-gold-500/5 border-gold-500/15':''}`;c.dataset.day=i;c.onclick=()=>toggleStar(i);c.innerHTML=`<iconify-icon icon="lucide:star" class="star-icon" width="20"></iconify-icon><span class="text-[8px] sm:text-[9px] ${col?'text-gold-400/60':'text-white/15'} font-medium">${i}</span>`;g.appendChild(c)}}
function toggleStar(day){const idx=starsState.indexOf(day);if(idx>-1){starsState.splice(idx,1)}else{starsState.push(day);starsState.sort((a,b)=>a-b);showToast(`Day ${day} completed! 🌟`)}localStorage.setItem('sm_stars',JSON.stringify(starsState));updateStarsUI()}
function resetStars(){starsState=[];localStorage.setItem('sm_stars',JSON.stringify(starsState));updateStarsUI();showToast('Tracker reset')}
function updateStarsUI(){document.querySelectorAll('.star-cell').forEach(c=>{const d=parseInt(c.dataset.day),v=starsState.includes(d);c.classList.toggle('colored',v);c.classList.toggle('bg-gold-500/5',v);c.classList.toggle('border-gold-500/15',v);const l=c.querySelector('span');if(l){l.classList.toggle('text-gold-400/60',v);l.classList.toggle('text-white/15',!v)}});document.getElementById('progressBar').style.width=(starsState.length/30*100)+'%';document.getElementById('progressText').textContent=starsState.length+'/30'}
function goToPage(page){if(page===currentPage||isAnimating||page<1||page>totalPages)return;isAnimating=true;const dir=page>currentPage?1:-1,old=document.getElementById('page'+currentPage),nxt=document.getElementById('page'+page);old.classList.remove('active');old.style.transform=dir>0?'translateX(-80px) scale(0.97)':'translateX(80px) scale(0.97)';old.style.opacity='0';nxt.style.transform=dir>0?'translateX(80px) scale(0.97)':'translateX(-80px) scale(0.97)';nxt.style.opacity='0';requestAnimationFrame(()=>{requestAnimationFrame(()=>{nxt.classList.add('active');nxt.style.transform='';nxt.style.opacity=''})});currentPage=page;updateNav();setTimeout(()=>{old.style.transform='';old.style.opacity='';isAnimating=false},800)}
function nextPage(){goToPage(currentPage+1)}function prevPage(){goToPage(currentPage-1)}
function updateNav(){document.querySelectorAll('.nav-dot').forEach((d,i)=>d.classList.toggle('active',i+1===currentPage));document.getElementById('pageCounter').textContent=String(currentPage).padStart(2,'0');document.getElementById('prevBtn').style.opacity=currentPage===1?'0.3':'1';document.getElementById('nextBtn').style.opacity=currentPage===totalPages?'0.3':'1'}
function showToast(text){const t=document.getElementById('toast');document.getElementById('toastText').textContent=text;t.style.opacity='1';t.style.transform='translate(-50%,0)';setTimeout(()=>{t.style.opacity='0';t.style.transform='translate(-50%,-16px)'},1800)}
document.addEventListener('keydown',(e)=>{if(e.key==='ArrowRight'||e.key==='ArrowDown')nextPage();if(e.key==='ArrowLeft'||e.key==='ArrowUp')prevPage()});
let touchStartX=0;document.addEventListener('touchstart',e=>{touchStartX=e.touches[0].clientX});document.addEventListener('touchend',e=>{const diff=touchStartX-e.changedTouches[0].clientX;if(Math.abs(diff)>60){diff>0?nextPage():prevPage()}});
updateNav();
